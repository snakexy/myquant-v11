## 5.1 基础工具函数

### 交易时间判断

```python
from datetime import datetime, time
from enum import Enum

class TradingPhase(Enum):
    """交易阶段枚举"""
    PRE_MARKET = "盘前集合竞价"
    MORNING = "上午交易"
    LUNCH_BREAK = "午间休息"
    AFTERNOON = "下午交易"
    CLOSE = "收盘"
    AFTER_HOURS = "盘后"
    NON_TRADING = "非交易时间"

def get_trading_phase() -> TradingPhase:
    """获取当前交易阶段"""
    now = datetime.now()
    current_time = now.time()
    weekday = now.weekday()

    # 周末是非交易时间
    if weekday >= 5:  # 周六、周日
        return TradingPhase.NON_TRADING

    # 定义关键时间点
    pre_market_start = time(9, 15)
    morning_start = time(9, 30)
    morning_end = time(11, 30)
    afternoon_start = time(13, 0)
    afternoon_end = time(15, 0)
    after_hours_end = time(24, 0)

    # 判断交易阶段
    if pre_market_start <= current_time < morning_start:
        return TradingPhase.PRE_MARKET
    elif morning_start <= current_time < morning_end:
        return TradingPhase.MORNING
    elif morning_end <= current_time < afternoon_start:
        return TradingPhase.LUNCH_BREAK
    elif afternoon_start <= current_time < afternoon_end:
        return TradingPhase.AFTERNOON
    elif afternoon_end <= current_time < after_hours_end:
        return TradingPhase.AFTER_HOURS
    else:
        return TradingPhase.NON_TRADING

def is_trading_time() -> bool:
    """判断是否为可交易时间（TdxQuant/XtQuant可用）"""
    phase = get_trading_phase()
    return phase in [
        TradingPhase.PRE_MARKET,
        TradingPhase.MORNING,
        TradingPhase.LUNCH_BREAK,
        TradingPhase.AFTERNOON,
        TradingPhase.CLOSE,
        TradingPhase.AFTER_HOURS
    ]

def is_market_open() -> bool:
    """判断市场是否开盘（9:30-15:00）"""
    phase = get_trading_phase()
    return phase in [TradingPhase.MORNING, TradingPhase.AFTERNOON]
```

## 5.2 数据服务基类

```python
from abc import ABC, abstractmethod
from typing import List, Dict, Any, Optional
from loguru import logger

class BaseDataService(ABC):
    """
    数据服务基类 - V5场景化服务基础

    提供统一的数据获取接口
    """

    def __init__(self):
        # 适配器初始化
        from data.adapters.tdxquant_adapter import TdxQuantAdapter
        from data.adapters.xtquant_adapter import XtQuantAdapter
        from data.adapters.pytdx_adapter import PyTdxAdapter
        from data.adapters.localdb_adapter import LocalDBAdapter

        self.tdxquant = TdxQuantAdapter()
        self.xtquant = XtQuantAdapter()
        self.pytdx2 = PyTdxAdapter(
            auto_select=True,
            auto_select_speed=True,
            enable_heartbeat=True,
            enable_cache=True
        )
        self.localdb = LocalDBAdapter()

    def is_trading_time(self) -> bool:
        """判断是否为交易时间"""
        return is_trading_time()

    def _get_adapter(self, data_type: str, use_primary: bool = True):
        """
        根据数据类型和时间获取适配器

        Args:
            data_type: 数据类型（snapshot/intraday_kline/history_kline/sector）
            use_primary: 是否使用优先级1

        Returns:
            适配器实例
        """
        is_trading = self.is_trading_time()

        # 板块数据路由
        if data_type == "sector":
            if is_trading:
                return self.tdxquant  # TdxQuant 6.99ms, 586板块
            else:
                return self.pytdx2   # PyTdx2 非交易时间唯一

        # 当天分钟线路由（TdxQuant不支持）
        elif data_type == "intraday_kline":
            if is_trading:
                return self.xtquant if use_primary else self.pytdx2
            else:
                return self.pytdx2   # 非交易时间唯一

        # 快照数据路由
        elif data_type == "snapshot":
            if is_trading:
                return self.tdxquant if use_primary else self.xtquant
            else:
                return self.pytdx2   # 非交易时间唯一

        # 历史K线路由
        elif data_type == "history_kline":
            return self.localdb if use_primary else self.pytdx2

        else:
            raise ValueError(f"未知数据类型: {data_type}")
```

## 5.3 板块数据实现

### 板块服务专用类

```python
class SectorDataService(BaseDataService):
    """
    板块数据服务 - 专用路由

    关键特性：
    - XtQuant完全不支持板块数据
    - TdxQuant交易时间最强（6.99ms, 586板块）
    - PyTdx2非交易时间唯一选择
    """

    def get_sectors(self, limit: int = 20) -> List[Dict]:
        """
        获取板块数据

        Args:
            limit: 返回板块数量

        Returns:
            板块数据列表

        板块路由：
        - 交易时间：TdxQuant（6.99ms, 586板块）🔥
        - 非交易时间：PyTdx2（~15ms，部分板块）🌟
        - ❌ XtQuant不支持
        """
        is_trading = self.is_trading_time()

        if is_trading:
            # 交易时间：TdxQuant绝对优势
            logger.info(f"使用 TdxQuant 获取板块数据（6.99ms, 586板块）")
            try:
                sectors = self.tdxquant.get_sectors(limit)
                return sectors
            except Exception as e:
                logger.warning(f"TdxQuant失败，降级到PyTdx2: {e}")
                sectors = self.pytdx2.get_sectors(limit)
                return sectors
        else:
            # 非交易时间：PyTdx2唯一可用
            logger.warning(f"非交易时间使用 PyTdx2 获取板块数据（板块数量有限）")
            return self.pytdx2.get_sectors(limit)

    def get_hot_sectors(self, limit: int = 20) -> List[Dict]:
        """
        获取热点板块

        实现：
        1. 获取所有板块行情
        2. 按涨跌幅排序
        3. 返回前N个

        板块路由：
        - 必须用TdxQuant（交易时间）
        - 或PyTdx2（非交易时间）
        - ❌ 不能用XtQuant
        """
        is_trading = self.is_trading_time()

        if is_trading:
            # 交易时间用TdxQuant
            all_sectors = self.tdxquant.get_sectors(limit=1000)
        else:
            # 非交易时间用PyTdx2
            all_sectors = self.pytdx2.get_sectors(limit=1000)

        # 按涨跌幅排序
        sorted_sectors = sorted(
            all_sectors,
            key=lambda x: x.get('change_pct', 0),
            reverse=True
        )

        return sorted_sectors[:limit]

    def get_sector_members(self, sector_code: str) -> List[str]:
        """
        获取板块成分股

        Args:
            sector_code: 板块代码

        Returns:
            成分股代码列表

        板块路由：
        - TdxQuant（交易时间）
        - PyTdx2（非交易时间）
        """
        is_trading = self.is_trading_time()

        if is_trading:
            return self.tdxquant.get_sector_members(sector_code)
        else:
            return self.pytdx2.get_sector_members(sector_code)
```

## 5.4 当天分钟线实现

### 分钟线服务专用类

```python
class IntradayKlineService(BaseDataService):
    """
    当天分钟线服务 - 专用路由

    关键特性：
    - ❌ TdxQuant不支持当天分钟线
    - XtQuant交易时间最优（0.90ms）
    - PyTdx2非交易时间唯一（10-19ms）
    """

    def get_minute_kline(self, symbols: List[str], period: str = '1min') -> Dict:
        """
        获取当天分钟K线

        Args:
            symbols: 股票代码列表
            period: 周期（1min/5min/15min/30min/60min）

        Returns:
            {symbol: kline_data}

        当天分钟线路由：
        - 交易时间：XtQuant → PyTdx2
        - 非交易时间：PyTdx2
        - ❌ TdxQuant不支持
        """
        is_trading = self.is_trading_time()
        result = {}

        if is_trading:
            # 交易时间：优先XtQuant（TdxQuant不支持）
            logger.info(f"使用 XtQuant 获取当天分钟线（{len(symbols)}只）")
            try:
                for symbol in symbols:
                    kline = self.xtquant.get_market_kline(symbol, period)
                    result[symbol] = kline
                return result
            except Exception as e:
                logger.warning(f"XtQuant失败，降级到PyTdx2: {e}")
                # 降级到PyTdx2
                for symbol in symbols:
                    kline = self.pytdx2.get_kline(symbol, period, count=240)
                    result[symbol] = kline
                return result
        else:
            # 非交易时间：只能用PyTdx2
            logger.warning(f"非交易时间使用 PyTdx2 获取当天分钟线")
            for symbol in symbols:
                kline = self.pytdx2.get_kline(symbol, period, count=240)
                result[symbol] = kline
            return result

    def get_realtime_kline(self, symbol: str, use_cache: bool = True):
        """
        获取实时分钟K线（带缓存）

        Args:
            symbol: 股票代码
            use_cache: 是否使用缓存

        Returns:
            K线数据

        缓存策略：
        - Redis缓存，TTL 60秒
        - 避免频繁请求
        """
        import redis
        import json

        # 检查缓存
        if use_cache:
            cache_key = f"kline:1min:{symbol}"
            try:
                cached = self.redis.get(cache_key)
                if cached:
                    logger.debug(f"缓存命中: {symbol}")
                    return json.loads(cached)
            except Exception:
                pass

        # 获取最新数据
        is_trading = self.is_trading_time()

        if is_trading:
            # 交易时间：优先XtQuant
            try:
                kline = self.xtquant.get_market_kline(symbol, '1min')
            except:
                kline = self.pytdx2.get_kline(symbol, '1min', count=240)
        else:
            # 非交易时间：只能用PyTdx2
            kline = self.pytdx2.get_kline(symbol, '1min', count=240)

        # 缓存60秒
        if use_cache:
            self.redis.setex(cache_key, 60, json.dumps(kline))

        return kline
```

## 5.5 快照数据实现

### 快照服务专用类

```python
class SnapshotService(BaseDataService):
    """
    快照服务 - 数量自适应路由

    关键特性：
    - TdxQuant 0.60ms 最快（<100股）
    - XtQuant 6ms 缓存（100-300股）
    - PyTdx2 分批（>300股或非交易时间）
    """

    def get_snapshot(self, symbols: List[str]) -> List[Dict]:
        """
        获取实时快照

        Args:
            symbols: 股票代码列表

        Returns:
            快照数据列表

        快照路由：
        - <100股：TdxQuant（0.60ms）🔥
        - 100-300股：XtQuant（6ms缓存）
        - >300股：PyTdx2分批
        - 非交易时间：PyTdx2
        """
        is_trading = self.is_trading_time()
        count = len(symbols)

        if is_trading:
            # 交易时间：按数量选择
            if count <= 100:
                # 少量股票用TdxQuant（0.60ms最快）
                logger.info(f"使用 TdxQuant 获取快照（{count}股）")
                return self.tdxquant.get_snapshot(symbols)
            elif count <= 300:
                # 中等数量用XtQuant（6ms缓存）
                logger.info(f"使用 XtQuant 获取快照（{count}股）")
                return self.xtquant.get_snapshot(symbols)
            else:
                # 大量股票用PyTdx2分批
                logger.info(f"使用 PyTdx2 分批获取快照（{count}股）")
                return self._batch_get_snapshot(symbols, batch_size=800)
        else:
            # 非交易时间：只能用PyTdx2
            logger.warning(f"非交易时间使用 PyTdx2 获取快照")
            return self._batch_get_snapshot(symbols, batch_size=800)

    def _batch_get_snapshot(self, symbols: List[str], batch_size: int = 800):
        """
        批量获取快照

        Args:
            symbols: 股票代码列表
            batch_size: 每批数量

        Returns:
            所有快照数据
        """
        all_results = []
        for i in range(0, len(symbols), batch_size):
            batch = symbols[i:i+batch_size]
            logger.debug(f"获取批次 {i//batch_size + 1}: {len(batch)}股")
            batch_results = self.pytdx2.get_snapshot(batch)
            all_results.extend(batch_results)

        return all_results
```

## 5.6 完整服务示例

### 监控服务完整实现

```python
class MonitoringService:
    """
    监控服务 - V5场景化服务

    功能：
    1. 热点板块发现
    2. 异常股票检测
    3. 实时行情监控

    路由策略：
    - 板块数据：TdxQuant(交易) → PyTdx2(非交易)
    - 快照数据：TdxQuant → XtQuant → PyTdx2
    """

    def __init__(self):
        # 初始化适配器
        from data.adapters.tdxquant_adapter import TdxQuantAdapter
        from data.adapters.xtquant_adapter import XtQuantAdapter
        from data.adapters.pytdx_adapter import PyTdxAdapter

        self.tdxquant = TdxQuantAdapter()
        self.xtquant = XtQuantAdapter()
        self.pytdx2 = PyTdxAdapter(
            auto_select=True,
            enable_heartbeat=True,
            enable_cache=True
        )

        # 初始化Redis缓存
        import redis
        self.redis = redis.Redis(host='localhost', port=6379, db=0)

    def is_trading_time(self) -> bool:
        """判断是否为交易时间"""
        from data.utils.trading_time_detector import TradingTimeDetectorV2
        detector = TradingTimeDetectorV2()
        phase = detector.get_trading_phase()
        return phase.name in ['MORNING', 'AFTERNOON']

    def get_hot_sectors(self, limit: int = 20):
        """
        获取热点板块

        板块路由：
        - 交易时间：TdxQuant（6.99ms, 586板块）🔥
        - 非交易时间：PyTdx2（~15ms）🌟
        """
        is_trading = self.is_trading_time()

        if is_trading:
            # 交易时间：TdxQuant绝对优势
            logger.info("使用 TdxQuant 获取热点板块（6.99ms, 586板块）")
            all_sectors = self.tdxquant.get_sectors(limit=1000)
        else:
            # 非交易时间：PyTdx2唯一可用
            logger.warning("非交易时间使用 PyTdx2 获取板块数据（板块数量有限）")
            all_sectors = self.pytdx2.get_sectors(limit=1000)

        # 按涨跌幅排序
        hot_sectors = sorted(
            all_sectors,
            key=lambda x: x.get('change_pct', 0),
            reverse=True
        )[:limit]

        return hot_sectors

    def get_hot_stocks(self, sector_code: str = None, limit: int = 50):
        """
        获取热点股票

        Args:
            sector_code: 板块代码（可选，限定板块内）
            limit: 返回数量

        快照路由：
        - <100股：TdxQuant（0.60ms）
        - 100-300股：XtQuant（6ms）
        - >300股：PyTdx2分批
        """
        # 如果指定了板块，先获取板块成分股
        if sector_code:
            if self.is_trading_time():
                symbols = self.tdxquant.get_sector_members(sector_code)
            else:
                symbols = self.pytdx2.get_sector_members(sector_code)
        else:
            # 获取所有股票
            symbols = self._get_all_symbols()
            limit = min(limit, 100)  # 限制数量

        # 根据数量选择数据源
        if len(symbols) <= 100:
            # 少量股票用TdxQuant（0.60ms最快）
            snapshots = self.tdxquant.get_snapshot(symbols)
        elif len(symbols) <= 300:
            # 中等数量用XtQuant（6ms缓存）
            snapshots = self.xtquant.get_snapshot(symbols)
        else:
            # 大量股票用PyTdx2分批
            snapshots = self._batch_get_snapshot(symbols[:500])

        # 按涨跌幅排序
        hot_stocks = sorted(
            snapshots,
            key=lambda x: x.get('change_pct', 0),
            reverse=True
        )[:limit]

        return hot_stocks
```
