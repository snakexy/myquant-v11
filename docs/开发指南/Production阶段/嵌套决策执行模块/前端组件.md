## 🔧 核心组件详解

### 1. NestedExecutionManager（嵌套执行管理器）

**路径**: `views/NestedExecutionManager.vue`

**功能**:
- 启动/停止嵌套执行系统
- 实时监控整体运行状态
- 管理子策略和订单

**Props**:
```typescript
interface Props {
  // 预设执行实例ID（查看模式）
  executionId?: string;
}
```

**State**:
```typescript
interface State {
  // 执行实例信息
  execution: NestedExecutionInstance | null;

  // UI状态
  activeTab: 'monitor' | 'orders' | 'performance' | 'logs';
  loading: boolean;

  // 实时数据
  levelStatuses: LevelStatus[];
  activeOrders: Order[];
  performanceMetrics: PerformanceMetrics;

  // WebSocket连接
  wsConnected: boolean;
}
```

**组件结构**:
```vue
<template>
  <div class="nested-execution-manager">
    <!-- 页面头部 -->
    <PageHeader
      title="嵌套决策执行系统"
      :status="execution?.status"
    >
      <template #actions>
        <el-button
          v-if="!execution"
          type="primary"
          @click="handleStartExecution"
        >
          <el-icon><VideoPlay /></el-icon>
          启动嵌套执行
        </el-button>
        <template v-else>
          <el-button
            type="danger"
            @click="handleStopExecution"
          >
            <el-icon><VideoPause /></el-icon>
            停止执行
          </el-button>
          <el-button @click="handleRefresh">
            <el-icon><Refresh /></el-icon>
            刷新
          </el-button>
        </template>
      </template>
    </PageHeader>

    <!-- 执行状态提示 -->
    <el-alert
      v-if="execution"
      :type="getStatusType(execution.status)"
      :closable="false"
    >
      <template #title>
        执行实例: {{ execution.execution_name }}
        ({{ execution.execution_id }})
        状态: {{ getStatusLabel(execution.status) }}
        运行时长: {{ execution.uptime_hours }}小时
      </template>
    </el-alert>

    <!-- 标签页 -->
    <el-tabs v-model="activeTab" v-if="execution">
      <!-- 监控标签 -->
      <el-tab-pane label="实时监控" name="monitor">
        <NestedExecutionMonitor
          :execution-id="execution.execution_id"
        />
      </el-tab-pane>

      <!-- 订单标签 -->
      <el-tab-pane label="订单监控" name="orders">
        <OrderExecutionMonitor
          :execution-id="execution.execution_id"
        />
      </el-tab-pane>

      <!-- 性能标签 -->
      <el-tab-pane label="性能指标" name="performance">
        <PerformanceMetricsDashboard
          :execution-id="execution.execution_id"
        />
      </el-tab-pane>

      <!-- 日志标签 -->
      <el-tab-pane label="实时日志" name="logs">
        <RealtimeLogsViewer
          :execution-id="execution.execution_id"
        />
      </el-tab-pane>
    </el-tabs>

    <!-- 启动配置对话框 -->
    <StartExecutionDialog
      v-model="showStartDialog"
      @confirm="handleConfirmStart"
    />
  </div>
</template>
```

interface Events {

## 📊 状态管理

### Pinia Store

**路径**: `store/nestedExecution.ts`

```typescript
import { defineStore } from 'pinia';

export const useNestedExecutionStore = defineStore('nestedExecution', {
  state: () => ({
    // 执行实例
    execution: null as NestedExecutionInstance | null,

    // 各级别状态
    levelStatuses: [] as LevelStatus[],

    // 活跃订单
    activeOrders: [] as Order[],

    // 性能指标
    performanceMetrics: null as PerformanceMetrics | null,

    // UI状态
    loading: false,
    activeTab: 'monitor',

    // WebSocket连接
    wsConnected: false,
    wsMessageQueue: [] as any[],
  }),

  getters: {
    // 获取运行中的实例
    runningExecution: (state) => {
      return state.execution?.status === 'running' ? state.execution : null;
    },

    // 获取活跃订单数
    activeOrdersCount: (state) => {
      return state.activeOrders.filter(o =>
        o.status === 'pending' || o.status === 'partial'
      ).length;
    },

    // 获取特定级别状态
    getLevelStatus: (state) => (level: number) => {
      return state.levelStatuses.find(ls => ls.level === level);
    },
  },

  actions: {
    // 启动嵌套执行
    async startExecution(config: NestedExecutionConfig) {
      this.loading = true;
      try {
        const response = await api.startNestedExecution(config);
        this.execution = response.data;
        await this.fetchLevelStatuses();
        return response.data;
      } finally {
        this.loading = false;
      }
    },

    // 停止嵌套执行
    async stopExecution(executionId: string, stopMode: string) {
      const response = await api.stopNestedExecution(executionId, { stop_mode: stopMode });
      if (this.execution?.execution_id === executionId) {
        this.execution.status = 'stopped';
      }
      return response.data;
    },

    // 获取级别状态
    async fetchLevelStatuses() {
      if (!this.execution) return;

      const response = await api.getNestedExecutionStatus(this.execution.execution_id);
      this.levelStatuses = Object.values(response.data.level_statuses);
      this.execution = response.data;
    },

    // 获取活跃订单
    async fetchActiveOrders() {
      if (!this.execution) return;

      const response = await api.getPendingOrders(this.execution.execution_id);
      this.activeOrders = response.data.orders;
    },

    // 获取性能指标
    async fetchPerformanceMetrics(timeRange?: string) {
      if (!this.execution) return;

      const response = await api.getPerformanceMetrics(
        this.execution.execution_id,
        { time_range: timeRange }
      );
      this.performanceMetrics = response.data;
    },

    // 动态调整子策略
    async adjustStrategy(adjustment: StrategyAdjustment) {
      const response = await api.adjustSubStrategy(adjustment);
      await this.fetchLevelStatuses();
      return response.data;
    },

    // 连接WebSocket
    connectWebSocket(executionId: string) {
      const ws = new WebSocket(`ws://localhost:8000/ws/nested/${executionId}`);

      ws.onopen = () => {
        this.wsConnected = true;
      };

      ws.onmessage = (event) => {
        const data = JSON.parse(event.data);
        this.wsMessageQueue.push(data);
        this.handleWebSocketMessage(data);
      };

      ws.onclose = () => {
        this.wsConnected = false;
        // 自动重连
        setTimeout(() => {
          if (this.execution?.status === 'running') {
            this.connectWebSocket(executionId);
          }
        }, 5000);
      };
    },

    // 处理WebSocket消息
    handleWebSocketMessage(data: any) {
      switch (data.type) {
        case 'level_status_update':
          this.updateLevelStatus(data.level, data.status);
          break;
        case 'order_update':
          this.updateOrder(data.order);
          break;
        case 'performance_update':
          this.performanceMetrics = data.metrics;
          break;
        case 'log':
          this.wsMessageQueue.push(data);
          break;
      }
    },

    // 更新级别状态
    updateLevelStatus(level: number, status: any) {
      const index = this.levelStatuses.findIndex(ls => ls.level === level);
      if (index !== -1) {
        this.levelStatuses[index] = { ...this.levelStatuses[index], ...status };
      }
    },

    // 更新订单
    updateOrder(order: Order) {
      const index = this.activeOrders.findIndex(o => o.order_id === order.order_id);
      if (index !== -1) {
        this.activeOrders[index] = order;
      } else {
        this.activeOrders.push(order);
      }
    },
  },
});
```
