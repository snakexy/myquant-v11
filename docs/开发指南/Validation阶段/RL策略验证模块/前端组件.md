### 2. ValidationTaskCreator（验证任务创建器）

**路径**: `components/rl/ValidationTaskCreator.vue`

**功能**:
- 选择RL模型和环境
- 配置验证参数
- 创建验证任务

**State**:
```typescript
interface State {
  form: {
    model_id: string;
    env_id: string;
    validation_config: ValidationConfig;
  };
  availableModels: RLModel[];
  availableEnvs: RLEnv[];
}
```

**组件结构**:
```vue
<template>
  <el-dialog v-model="visible" title="创建RL验证任务">
    <el-form :model="form">
      <el-form-item label="RL模型">

            v-for="model in availableModels"
            :key="model.model_id"
            :label="model.model_name"
            :value="model.model_id"
          />
        </el-select>
      </el-form-item>

      <el-form-item label="环境">
        <el-select v-model="form.env_id">
          <el-option
            v-for="env in availableEnvs"
            :key="env.env_id"
            :label="env.env_name"
            :value="env.env_id"
          />
        </el-select>
      </el-form-item>

      <el-form-item label="起始时间">
        <el-date-picker v-model="form.validation_config.start_time" type="date" />
      </el-form-item>

      <el-form-item label="结束时间">
        <el-date-picker v-model="form.validation_config.end_time" type="date" />
      </el-form-item>

      <el-form-item label="训练回合数">
        <el-input-number v-model="form.validation_config.n_episodes" :min="10" />
      </el-form-item>

      <el-form-item label="确定性执行">
        <el-switch v-model="form.validation_config.deterministic" />
      </el-form-item>
    </el-form>

    <template #footer>
      <el-button @click="visible = false">取消</el-button>
      <el-button type="primary" @click="handleCreate">创建</el-button>
```

## 🔄 API调用封装

```typescript
// frontend/src/api/rl-validation.ts
export const rlValidationApi = {
  // 创建验证任务
  createValidation: (data: {
    model_id: string;
    env_id: string;
    validation_config: ValidationConfig;
  }) => {
    return request.post('/api/v1/validation/rl/validate/create', data);
  },

  // 查询验证进度
  getProgress: (validationId: string) => {
    return request.get(`/api/v1/validation/rl/validate/progress/${validationId}`);
  },

  // 获取验证结果
  getResult: (validationId: string) => {
    return request.get(`/api/v1/validation/rl/validate/result/${validationId}`);
  },

  // 创建策略对比
  createComparison: (data: {
    strategies: Strategy[];
    comparison_config: ComparisonConfig;
  }) => {
    return request.post('/api/v1/validation/rl/compare/create', data);
  },

  // 获取对比结果
  getComparisonResult: (comparisonId: string) => {
    return request.get(`/api/v1/validation/rl/compare/result/${comparisonId}`);
  }
};
```
