## 数据模型定义

### 1. 策略配置模型

```python
from pydantic import BaseModel, Field
from typing import Literal, Optional, Dict, Any
from datetime import datetime

class StrategyConfig(BaseModel):
    """策略配置模型"""

    strategy_type: Literal["TopkDropoutStrategy", "EnhancedIndexingStrategy"] = Field(
        description="策略类型"
    )

    params: Dict[str, Any] = Field(
        default={},
        description="策略参数"
    )

    created_at: datetime = Field(
        default_factory=datetime.now,
        description="创建时间"
    )

    updated_at: Optional[datetime] = Field(
        default=None,
        description="更新时间"
    )

    class Config:
        json_schema_extra = {
            "example": {
                "strategy_type": "TopkDropoutStrategy",
                "params": {
                    "topk": 50,
                    "n_drop": 5
                }
            }
        }
```

### 2. TopkDropoutStrategy 配置

```python
class TopkDropoutConfig(BaseModel):
    """TopkDropout策略配置"""

    topk: int = Field(
        ge=10, le=100, default=50,
        description="持有股票数量"
    )

    n_drop: int = Field(
        ge=1, le=20, default=5,
        description="每日卖出数量"
    )

    signal: Optional[str] = Field(
        default=None,
        description="预测信号ID"
    )
```

### 3. EnhancedIndexingStrategy 配置

```python
class EnhancedIndexingConfig(BaseModel):
    """增强指数化策略配置"""

    risk_thld: float = Field(

        description="风险容忍度"

        description="换手率限制"

        default="SH000300",
        description="基准指数"

```

### 4. 投资组合模型

```python
class PortfolioStock(BaseModel):
    """投资组合中的单个股票"""

    instrument: str = Field(
        description="股票代码"
    )

    score: float = Field(
        description="预测分数"
    )

    weight: float = Field(
        ge=0.0, le=1.0, description="权重"
    )

    action: Literal["buy", "sell", "hold"] = Field(
        description="操作类型"
    )

class Portfolio(BaseModel):
    """投资组合模型"""

    portfolio_id: str = Field(
        description="投资组合ID"
    )

    stocks: List[PortfolioStock] = Field(
        description="股票列表"
    )

    summary: Dict[str, Any] = Field(
        description="汇总信息"
    )

    created_at: datetime = Field(
        default_factory=datetime.now,
        description="创建时间"
    )
```

### 5. 回测结果模型

```python
class BacktestMetrics(BaseModel):
    """回测指标"""

    annual_return: float = Field(
        description="年化收益率"
    )

    sharpe_ratio: float = Field(
        description="夏普比率"
    )

    max_drawdown: float = Field(
        description="最大回撤"
    )

    win_rate: float = Field(
        ge=0.0, le=1.0,
        description="胜率"
    )

class BacktestResult(BaseModel):
    """回测结果"""

    backtest_id: str = Field(
        description="回测ID"
    )

    metrics: BacktestMetrics = Field(
        description="回测指标"
    )

    positions: List[Dict[str, Any]] = Field(
        default=[],
        description="持仓明细"
    )

    start_date: str = Field(
        description="开始日期"
    )

    end_date: str = Field(
        description="结束日期"
    )
```

### 6. 参数优化评估结果

```python
class OptimizationMetrics(BaseModel):
```

## API 请求/响应模型

### 生成投资组合请求

```python
class GeneratePortfolioRequest(BaseModel):
    """生成投资组合请求"""

    signal_id: str = Field(
        description="预测信号ID"
    )

    strategy_type: Literal["TopkDropoutStrategy", "EnhancedIndexingStrategy"] = Field(
        default="TopkDropoutStrategy",
        description="策略类型"
    )

    strategy_params: Optional[Dict[str, Any]] = Field(
        default=None,
        description="策略参数（如不提供则使用默认）"
    )

    date: str = Field(
        description="交易日期"
    )
```
