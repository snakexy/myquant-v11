    states = []
    for i in range(len(kline_data)):
        if i < 20:
            continue  # 前20天数据不足，跳过

        current_price = kline_data['close'].iloc[i]

        prev_ma20 = ma20.iloc[i-1]

        # 归一化
        norm_price = current_price / prev_ma20

        state = np.array([
            norm_price,
            self.position_info['shares'] / 100,
            self.position_info['cash'] / 10000,
            prev_ma5 / 200,
            prev_ma20 / 200,
            kline_data['RSI'].iloc[i] / 100
        ])
        states.append(state)

    # 真实回测
    total_return = self._backtest_with_real_data(
        strategy_params,

# 修改 RealDataTradingEnv 类

class RealDataTradingEnv:
    def __init__(self, symbol: str, manager):
        self.symbol = symbol
        self.manager = manager

        # ✅ 初始化时获取真实K线数据
        self._load_historical_data()

    def _load_historical_data(self):
        """从UnifiedDataManager加载历史数据"""
        end_date = datetime.now()
        start_date = end_date - timedelta(days=180)

        result = self.manager.get_data(
            data_type=DataType.KLINE,
            symbols=[self.symbol],
            context={
                'period': 'day',
                'start_date': start_date.strftime('%Y-%m-%d'),
                'end_date': end_date.strftime('%Y-%m-%d'),
                'adjust_type': 'front'
            }
        )

        if result and self.symbol in result:
            self.kline_data = result[self.symbol]
            logger.success(f"✅ 加载真实K线数据: {len(self.kline_data)}条")
        else:
            logger.warning(f"⚠️ 无法获取K线数据，使用模拟数据")
            self.kline_data = self._generate_mock_data()

    def reset(self):
        """重置环境到初始状态"""
        # 使用真实数据初始化状态
        if self.kline_data is not None:
            last_close = self.kline_data['close'].iloc[-1]
            last_ma5 = self.kline_data['MA5'].iloc[-1]

            self.state = np.array([
                last_close / last_ma5,
                0,  # 无持仓
                1.0,  # 100%现金
                0.5,  # MA指标位置
                0.5  # RSI指标位置
            ])

        return self.state

    def step(self, action):
        """执行一步交易"""
        # 真实数据回测逻辑
        current_idx = self.current_step

        # 0=持有, 1=买入, 2=卖出
        if action == 1:  # 买入
            self.position_info['shares'] += 100
            self.position_info['cash'] -= self.kline_data['close'].iloc[current_idx]
        elif action == 2:  # 卖出
            self.position_info['cash'] += self.kline_data['close'].iloc[current_idx] * self.position_info['shares']
            self.position_info['shares'] = 0

        # 计算真实奖励
        if self.current_step < len(self.kline_data) - 1:
            next_price = self.kline_data['close'].iloc[self.current_step + 1]
            reward = (next_price - self.kline_data['close'].iloc[current_idx])

            # 奖励归一化
            reward = reward / self.kline_data['close'].iloc[current_idx]

        # 更新状态
        self.current_step += 1
        return self._get_state(), reward, done, {}

# 文件: backend/api/v1/research/rl_router.py (新建)

from fastapi import APIRouter
from ai_strategy_engine.strategy_optimizer import StrategyOptimizer

router = APIRouter(prefix="/api/v1/research", tags=["RL策略优化"])

optimizer = StrategyOptimizer()

@router.post("/rl/train")
async def train_rl_strategy(request: TrainRequest):
    """
    RL策略训练API

    功能:
    - 接收训练参数（学习率、训练轮数等）
    - 调用strategy_optimizer.evaluate_strategy()使用真实K线数据
    - 返回训练结果（损失、性能指标等）
    """
    try:
        # 从UnifiedDataManager获取K线数据
        from data.unified_data_manager import get_unified_data_manager, DataType

        manager = get_unified_data_manager()

        symbol = request.symbol
        end_date = datetime.now()
        start_date = end_date - timedelta(days=180)

        result = manager.get_data(
            data_type=DataType.KLINE,
            symbols=[symbol],
            context={
                'period': 'day',
                'start_date': start_date.strftime('%Y-%m-%d'),
                'end_date': end_date.strftime('%Y-%m-%d'),
                'adjust_type': 'front'
            }
        )

        if not result or symbol not in result:
            raise HTTPException(status_code=404, detail=f"无法获取{symbol}的K线数据")

        kline_data = result[symbol]

        # 使用真实数据评估策略
        eval_result = optimizer.evaluate_strategy(
            strategy_params=request.strategy_params,
            kline_data=kline_data
        )

        return {
            "success": True,
            "data": {
                "sharpe_ratio": eval_result['sharpe_ratio'],
                "max_drawdown": eval_result['max_drawdown'],
                "total_return": eval_result['total_return'],
                "data_points": eval_result['data_points']
            },
            "message": "真实数据回测完成"
        }

    except Exception as e:
        logger.error(f"RL策略训练失败: {e}")
        raise HTTPException(status_code=500, detail=str(e))
