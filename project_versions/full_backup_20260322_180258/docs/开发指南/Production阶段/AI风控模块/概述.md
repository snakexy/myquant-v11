def adjust_threshold_by_volatility(current_volatility: float) -> float:
    """
    根据波动率调整阈值

    - 高波动 (vol > 0.3): 收紧阈值，更保守
    - 中等波动 (0.2 < vol < 0.3): 适度调整
    - 低波动 (vol < 0.2): 可适当放宽
    """
    if current_volatility > 0.3:
        return BASE_THRESHOLD * 0.85  # 收紧15%
    elif current_volatility > 0.2:
        return BASE_THRESHOLD * 0.95  # 收紧5%
    else:
        return BASE_THRESHOLD  # 保持不变

### 3. VaR/CVaR分析
```python
def calculate_var_cvar(returns: np.ndarray, confidence: float = 0.95) -> tuple:
    """
    计算VaR和CVaR

    Args:
        returns: 历史收益率序列
        confidence: 置信水平

    Returns:
        var: Value at Risk
        cvar: Conditional VaR (Expected Shortfall)
    """
    var = np.percentile(returns, (1 - confidence) * 100)
    cvar = returns[returns <= var].mean()
    return var, cvar
```
