```typescript
// 获取预测
const handlePredict = async () => {
  predicting.value = true;
  try {
    const response = await api.post(
      '/api/v1/production/online/predict',
      {
        instruments: form.value.instruments,
        datetime: form.value.predictionDate
      }
    );
    predictions.value = Object.entries(response.data.predictions)
      .map(([instrument, score]) => ({ instrument, score }))
      .sort((a, b) => b.score - a.score)
      .map((item, index) => ({ ...item, rank: index + 1 }));
  } finally {
    predicting.value = false;
  }
};

// 批量预测
const handleBatchPredict = async () => {
  batchPredicting.value = true;
  try {
    const response = await api.post(
      '/api/v1/production/online/predict/batch',
      {
        market: batchForm.value.market,
        datetime: new Date().toISOString().split('T')[0],
        filter: {
          min_market_cap: batchForm.value.minMarketCap,
          max_st: batchForm.value.maxSt
        }
      }
    );
    batchPredictions.value = response.data.predictions;
  } finally {
    batchPredicting.value = false;
  }
};
```

```typescript
// 执行例行更新
const handleExecute = async () => {
  executing.value = true;
  currentStepIndex.value = 0;
  stepsCompleted.value = [];

  try {
    const response = await api.post(
      '/api/v1/production/online/routine/execute',
      {
        cur_time: config.value.curTime,
        steps: Object.fromEntries(
          selectedSteps.value.map(step => [step, true])
        ),
        task_kwargs: { segments: config.value.segments },
        model_kwargs: { loss: config.value.loss },
        signal_kwargs: { prepare_func: config.value.prepareFunc }
      }
    );

    routineResult.value = response.data;

    // 模拟进度更新
    for (const step of selectedSteps.value) {
      currentStep.value = step;
      await new Promise(resolve => setTimeout(resolve, 1000));
      stepsCompleted.value.push(step);
      currentStepIndex.value++;
    }
  } finally {
    executing.value = false;
  }
};

// 获取更新历史
const fetchHistory = async () => {
```

```typescript
import request from '@/utils/request';

export const onlineModelApi = {

  },

  // 获取实时预测
  getPredictions: (data: {
    instruments: string[];
    datetime: string;
```

## 📊 WebSocket实时通信

### 连接管理

```typescript
// frontend/src/utils/websocket.ts
class OnlineModelWebSocket {
  private ws: WebSocket | null = null;
  private url: string;

  constructor(url: string) {
    this.url = url;
  }

  connect() {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      console.log('WebSocket连接成功');
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleMessage(data);
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket错误:', error);
    };

    this.ws.onclose = () => {
      console.log('WebSocket连接关闭');
      // 自动重连
      setTimeout(() => this.connect(), 5000);
    };
  }

  private handleMessage(data: any) {
    switch (data.type) {
      case 'performance_update':
        // 更新性能指标
        break;
      case 'model_switched':
        // 模型切换通知
        break;
      case 'alert':
        // 性能告警
        break;
      case 'routine_completed':
        // 例行更新完成
        break;
    }
  }

  send(data: any) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data));
    }
  }

  disconnect() {
    this.ws?.close();
  }
}
```
