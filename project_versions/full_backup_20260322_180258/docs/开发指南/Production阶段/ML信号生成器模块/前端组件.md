### 1. MLSignalDisplay - ML信号展示组件

```vue
<!-- components/trading/MLSignalDisplay.vue -->

<template>
  <div class="ml-signal-display">
    <!-- 预测评分展示 -->
    <div class="score-indicator">
      <div class="score-bar">
        <div
          :style="{ width: `${prediction_score * 100}%` }"
          :class="getScoreClass(prediction_score)"
        >
          <span class="score-text">
            ML评分: {{ (prediction_score * 100).toFixed(0) }}
          </span>
        </div>
      </div>
      <div class="confidence-info">
        置信度: {{ (confidence * 100).toFixed(0) }}%
      </div>
    </div>

    <!-- 信号类型 -->
    <el-tag :type="getSignalTagType(signal_type)">
      {{ getSignalText(signal_type) }}
    </el-tag>

    <!-- 详细信息 -->
    <div class="signal-details">
      <p>当前价: {{ current_price.toFixed(2) }}</p>
      <p>目标价: {{ target_price.toFixed(2) }}</p>
      <p class="reason">原因: {{ reason }}</p>
      <p>模型: {{ ml_model_id }}</p>
    </div>
  </div>
</template>

<script setup lang="ts">
interface Props {
  prediction_score: number;
  confidence: number;
  signal_type: string;
  current_price: number;
  target_price: number;
  reason: string;
  ml_model_id: string;
}

const props = defineProps<Props>();

const getScoreClass = (score: number) => {
  if (score >= 0.8) return 'score-strong';
  if (score >= 0.6) return 'score-normal';
  if (score >= 0.4) return 'score-weak';
  return 'score-sell';
};

const getSignalTagType = (type: string) => {
  const map: Record<string, string> = {
    BUY: 'success',
    SELL: 'danger',
    HOLD: 'warning',
    WATCH: 'info'
  };
  return map[type] || 'info';
};

const getSignalText = (type: string) => {
  const map: Record<string, string> = {
    BUY: '买入',
    SELL: '卖出',
    HOLD: '持有',
    WATCH: '观望'
  };
  return map[type] || type;
};
</script>

<style scoped>
.ml-signal-display {
  padding: 16px;
  border-radius: 8px;
  background: #f5f7fa;
}

.score-bar {
  height: 32px;
  background: #e0e0e0;
  border-radius: 16px;
  overflow: hidden;
}

.score-bar > div {
  height: 100%;
  display: flex;
  align-items: center;
  padding: 0 16px;
  color: white;
  font-weight: bold;
}

.score-strong { background: #67c23a; }
.score-normal { background: #409eff; }
.score-weak { background: #e6a23c; }
.score-sell { background: #f56c6c; }

.signal-details {
  margin-top: 16px;
  font-size: 14px;
  color: #606266;
}

.reason {
  color: #909399;
  font-style: italic;
}
</style>
```

<!-- components/trading/MLRiskDashboard.vue -->

        </div>
        <div class="risk-metrics">
          <p>预测集中度: {{ (riskMetrics.concentration_ratio * 100).toFixed(0) }}%</p>
          <p>上涨预测: {{ riskMetrics.up_count }} / 下跌预测: {{ riskMetrics.down_count }}</p>
          <p>平均置信度: {{ (riskMetrics.avg_confidence * 100).toFixed(0) }}%</p>

        :title="riskMetrics.recommendation"

      />
    </div>

    <!-- 风险趋势图 -->
    <div class="risk-chart">
      <el-button @click="loadRiskTrend">加载风险趋势</el-button>
