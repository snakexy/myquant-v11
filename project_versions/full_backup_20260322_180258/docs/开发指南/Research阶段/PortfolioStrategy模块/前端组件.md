## 1. 策略配置面板

### 组件路径
```
frontend/src/views/research/components/PortfolioStrategyConfig.vue
```

### 功能说明
- 选择策略类型（TopkDropout / EnhancedIndexing）
- 配置策略参数
- 保存/加载配置
- 预览策略效果

### 组件设计

```vue
<template>
  <div class="portfolio-strategy-config">
    <el-card title="投资组合策略配置">
      <!-- 策略类型选择 -->
      <el-form-item label="策略类型">
        <el-select v-model="config.strategy_type">
          <el-option
            label="Topk-Drop策略"
            value="TopkDropoutStrategy"
          />
          <el-option
            label="增强指数化策略"
            value="EnhancedIndexingStrategy"
          />
        </el-select>
      </el-form-item>

      <!-- TopkDropout参数 -->
      <template v-if="config.strategy_type === 'TopkDropoutStrategy'">
        <el-form-item label="持有股票数 (topk)">
          <el-input-number
            v-model="config.params.topk"
            :min="10"
            :max="100"
          />
        </el-form-item>
        <el-form-item label="每日卖出数 (n_drop)">
          <el-input-number
            v-model="config.params.n_drop"
            :min="1"
            :max="20"
          />
        </el-form-item>
      </template>

      <!-- EnhancedIndexing参数 -->
      <template v-if="config.strategy_type === 'EnhancedIndexingStrategy'">
        <el-form-item label="风险容忍度">
          <el-slider v-model="config.params.risk_thld" :max="0.5" />
        </el-form-item>
        <el-form-item label="换手率限制">
          <el-slider v-model="config.params.turn_rate" :max="1" />
        </el-form-item>
      </template>

      <!-- 操作按钮 -->
      <el-form-item>
        <el-button @click="saveConfig">保存配置</el-button>
        <el-button @click="loadConfig">加载配置</el-button>
        <el-button @click="generatePortfolio">生成投资组合</el-button>
      </el-form-item>
    </el-card>
  </div>
</template>
```

## API 调用封装

### TypeScript API

```typescript
// frontend/src/api/portfolioStrategy.ts

import { request } from '@/utils/request';

export interface StrategyConfig {
  strategy_type: 'TopkDropoutStrategy' | 'EnhancedIndexingStrategy';
  params: Record<string, any>;
}

export interface PortfolioStock {
  instrument: string;
  score: number;
  weight: number;
  action: 'buy' | 'sell' | 'hold';
}

export interface Portfolio {
  portfolio_id: string;
  stocks: PortfolioStock[];
  summary: {
    total_stocks: number;
    buy_count: number;
    sell_count: number;
    hold_count: number;
  };
}

export interface BacktestMetrics {
  annual_return: number;
  sharpe_ratio: number;
  max_drawdown: number;
  win_rate: number;
}

// 获取策略列表
export function getStrategies() {
  return request({
    url: '/api/v1/research/portfolio/strategies',
    method: 'get'
  });
}

// 保存策略配置
export function saveConfig(config: StrategyConfig) {
  return request({
    url: '/api/v1/research/portfolio/config',
    method: 'post',
    data: config
  });
}

// 生成投资组合
export function generatePortfolio(signalId: string, strategyType: string, params: Record<string, any>, date: string) {
  return request({
    url: '/api/v1/research/portfolio/generate',
    method: 'post',
    data: {
      signal_id: signalId,
      strategy_type: strategyType,
      strategy_params: params,
      date: date
    }
  });
}

// 策略回测
export function backtest(signalId: string, strategyType: string, params: Record<string, any>, startDate: string, endDate: string) {
  return request({
    url: '/api/v1/research/portfolio/backtest',
    method: 'post',
    data: {
      signal_id: signalId,
      strategy_type: strategyType,
      strategy_params: params,
      start_date: startDate,
      end_date: endDate
    }
  });
}
```
