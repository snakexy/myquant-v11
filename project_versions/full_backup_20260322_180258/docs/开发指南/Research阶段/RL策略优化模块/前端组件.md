**Props**:
```typescript

  rewardType?: 'returns' | 'sharpe' | 'sortino' | 'custom';
  customFormula?: string;

}
```

**核心功能**:
- 预设奖励类型选择
- 自定义公式编辑器
- 惩罚项配置
- 实时公式验证
- 奖励分布预览

**使用示例**:
```vue
<template>
  <RewardFunctionEditor
    v-model:reward-type="rewardType"
    v-model:custom-formula="customFormula"
    v-model:penalty="penaltyConfig"
    @reward-change="handleRewardChange"
  />
</template>
```

### 3. RLTrainingDashboard - RL训练仪表板

**功能**: 实时监控RL训练过程

**Props**:
```typescript

  taskId: string;
  refreshInterval?: number;  // 刷新间隔(秒)
}
```

**状态管理**:
```typescript

  task: RLTrainingTask | null;
  progress: TrainingProgress;
  performance: PerformanceMetrics;
  trainingCurve: TrainingCurve;
  isRunning: boolean;
}
```

**子组件**:
- `TrainingProgressBar` - 训练进度条
- `EpisodeRewardChart` - 回合奖励曲线
- `PerformanceMetricsPanel` - 性能指标面板
- `TrainingControlPanel` - 训练控制面板

**界面示例**:
```vue
<template>
  <div class="rl-training-dashboard">
    <!-- 进度概览 -->
    <TrainingProgressBar
      :current="progress.current_timestep"
      :total="progress.total_timesteps"
      :percentage="progress.percentage"
    />

    <!-- 实时性能 -->
    <PerformanceMetricsPanel
      :sharpe="performance.sharpe"
      :returns="performance.returns"
      :max-drawdown="performance.max_drawdown"
    />

    <!-- 训练曲线 -->
    <EpisodeRewardChart
      :data="trainingCurve.episode_rewards"
      :smooth="true"
    />

    <!-- 控制按钮 -->
    <TrainingControlPanel
      :is-running="isRunning"
      @pause="pauseTraining"
      @resume="resumeTraining"
      @stop="stopTraining"
    />
  </div>
</template>
```

## 🔄 状态管理方案

### Pinia Store定义

```typescript
// stores/rlStrategy.ts
import { defineStore } from 'pinia';

export const useRLStrategyStore = defineStore('rlStrategy', {
  state: () => ({
    // 环境
    environments: [] as RLEnvironment[],
    currentEnv: null as RLEnvironment | null,

    // 奖励函数
    rewardFunctions: [] as RewardFunction[],
    currentReward: null as RewardFunction | null,

    // 训练任务
    trainingTasks: [] as RLTrainingTask[],
    currentTask: null as RLTrainingTask | null,
    trainingProgress: null as TrainingProgress | null,

    // 模型
    models: [] as RLModel[],
    currentModel: null as RLModel | null,

    // UI状态
    loading: false,
    error: null as string | null,
  }),

  actions: {
    // 环境管理
    async createEnvironment(config: EnvConfig) {
      this.loading = true;
      try {
        const env = await api.rl.createEnv(config);
        this.environments.push(env);
        return env;
      } finally {
        this.loading = false;
      }
    },

    // 奖励函数
    async configureReward(config: RewardConfig) {
      const reward = await api.rl.configureReward(config);
      this.rewardFunctions.push(reward);
      this.currentReward = reward;
      return reward;
    },

    // 训练
    async createTrainingTask(config: TrainingConfig) {
      const task = await api.rl.createTraining(config);
      this.trainingTasks.push(task);
      this.currentTask = task;
      return task;
    },

    async getTrainingProgress(taskId: string) {
      const progress = await api.rl.getTrainingProgress(taskId);
      this.trainingProgress = progress;
      return progress;
    },

    // 模型管理
    async saveModel(taskId: string, modelName: string) {
      const model = await api.rl.saveModel(taskId, modelName);
      this.models.push(model);
      return model;
    },

    // 评估
    async evaluateModel(modelId: string, envId: string) {
      const results = await api.rl.evaluateModel(modelId, envId);
      return results;
    },
  },
});
```
