# 1. 从UnifiedDataManager获取数据
from data.unified_data_manager import get_unified_data_manager, DataType
manager = get_unified_data_manager()
result = manager.get_data(data_type=DataType.KLINE, symbols=[self.symbol], ...)
historical_data = result[self.symbol]

# 2. 计算技术指标
ma5 = historical_data['close'].rolling(window=5).mean()
ma20 = historical_data['close'].rolling(window=20).mean()

# 3. 回测循环
for i in range(len(historical_data)):
    current_price = historical_data['close'].iloc[i]
    prev_ma20 = ma20.iloc[i-1]

    # 简化的交易逻辑
    if current_price > prev_ma20:
        action = 1  # 买入
        reward = (current_price - prev_price) / prev_price
    elif current_price < prev_ma20:
        action = 2  # 卖出
        reward = (prev_price - current_price) / prev_price
    else:
        action = 0  # 持有

    # 计算总收益率
total_return = sum(rewards[1:]) / len(rewards)

# 4. 计算评估指标
if evaluation_target == 'sharpe_ratio':
    sharpe_ratio = (mean_return * 252) / np.std(rewards[1:])
    score = max(0, sharpe_ratio)
elif evaluation_target == 'max_drawdown':
    max_drawdown_ratio = drawdown / peak if peak > 0 else 0
    score = -max_drawdown_ratio  # 越小越好
else:
    score = total_return
