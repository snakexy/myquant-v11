**优化方案**:

```python
from functools import lru_cache
from datetime import datetime, timedelta

class LocalCache:
    """本地缓存管理"""
    def __init__(self):
        self._xdxr_cache = {}  # 除权信息缓存
        self._sector_cache = {}  # 板块数据缓存
        self._cache_time = {}  # 缓存时间

    def get_xdxr_info(self, symbol):
        """获取除权信息（带缓存）"""
        cache_key = f"xdxr_{symbol}"

        # 缓存有效期：除权信息1天，板块数据1小时
        if cache_key in self._cache_time:
            if datetime.now() - self._cache_time[cache_key] < timedelta(days=1):
                return self._xdxr_cache.get(cache_key)

        # 缓存过期，重新获取
        data = self._api.get_xdxr_info(symbol)
        self._xdxr_cache[cache_key] = data
        self._cache_time[cache_key] = datetime.now()
        return data
```

**预期效果**: 除权信息、板块数据请求减少 90%

### 4. 请求合并优化 (中优先级)

**当前问题**: 多个小批量请求未合并

**优化方案**:
```python
class RequestBatcher:
    """请求批处理"""
    def __init__(self, max_batch_size=800, max_wait_time=0.1):
        self._pending = []
        self._max_batch_size = max_batch_size
        self._max_wait_time = max_wait_time  # 秒

    def add_request(self, symbols, callback):
        """添加请求到批处理队列"""
        self._pending.append((symbols, callback))

        # 达到批量大小或超时，执行请求
        if len(self._pending) >= self._max_batch_size:
            self._flush()
        elif len(self._pending) == 1:
            # 启动超时定时器
            threading.Timer(self._max_wait_time, self._flush).start()

    def _flush(self):
        """执行批处理请求"""
        if not self._pending:
            return

        # 合并所有请求的股票代码
        all_symbols = []
        callbacks = []
        for symbols, callback in self._pending:
            all_symbols.extend(symbols)
            callbacks.append(callback)

        # 一次性请求
        quotes = self.adapter.get_realtime_quote(all_symbols)

        # 分发结果
        idx = 0
        for callback in callbacks:
            batch_size = len(self._pending[idx][0])
            callback(quotes[idx:idx+batch_size])
            idx += batch_size

        self._pending.clear()
```

**预期效果**: 高频请求场景下减少 50% 网络往返

```python
import asyncio

class AsyncPyTdxAdapter:
    """异步适配器"""

    async def get_realtime_quote_async(self, symbols: List[str]):
        """异步获取实时快照"""
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(
            self._executor,
            self._sync_get_realtime_quote,
            symbols
        )

    async def get_multiple_quotes_async(self, symbols_groups: List[List[str]]):
        """并发获取多组股票数据"""
        tasks = [self.get_realtime_quote_async(group) for group in symbols_groups]
        results = await asyncio.gather(*tasks)
        return results
```

```python
class AsyncPyTdxAdapter(PyTdxAdapter):
```
