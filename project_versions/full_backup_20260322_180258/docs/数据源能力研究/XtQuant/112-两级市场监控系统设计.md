# 两级市场监控系统设计

> **创建日期**: 2026-02-05
> **目的**: 解决订阅数量限制，实现高效的市场监控

---

## 核心思想

**问题**: 单实例订阅上限300-500只，无法监控全市场5186只股票

**解决方案**: 两级监控
1. **Level 1（全A扫描）**: 未订阅，低频（60秒），发现异常
2. **Level 2（重点监控）**: 已订阅，高频（3秒），密切关注

---

## 系统架构

```
全市场（5186只）
    ↓
Level 1: 全A扫描器（未订阅）
    - 频率: 60秒/次
    - 速度: ~550ms/次
    - 作用: 发现异常股票
    ↓
    ↓ 发现异常（涨停、跌停、放量等）
    ↓
Level 2: 重点监控器（已订阅）
    - 频率: 3秒/次
    - 速度: ~10ms/次
    - 容量: 300-500只
    - 作用: 密切关注 + 实时追踪
```

## Level 1: 全A扫描器

### 设计

```python
from xtquant import xtdata
import time

class FullMarketScanner:
    """全A市场扫描器"""

    def __init__(self):
        # 获取所有A股
        self.all_stocks = xtdata.get_stock_list_in_sector('沪深A股')
        print(f"[全A扫描] 股票总数: {len(self.all_stocks)}")

        # 异常股票池
        self.anomaly_watchlist = set()

    def scan_market(self, interval=60):
        """
        扫描全市场

        Args:
            interval: 扫描间隔（秒），默认60秒
        """
        print(f"[启动] 全A扫描器，间隔: {interval}秒")

        while True:
            # 分批扫描（每批500只）
            batch_size = 500
            total_anomalies = 0

            for i in range(0, len(self.all_stocks), batch_size):
                batch = self.all_stocks[i:i+batch_size]

                # 获取快照（未订阅，速度慢）
                start = time.time()
                quotes = xtdata.get_full_tick(batch)
                elapsed = (time.time() - start) * 1000

                # 检测异常
                batch_anomalies = self._detect_anomalies(quotes)
                total_anomalies += len(batch_anomalies)

                # 添加到异常池
                for anomaly in batch_anomalies:
                    self.anomaly_watchlist.add(anomaly['symbol'])
                    print(f"[异常] {anomaly['symbol']}: {anomaly['type']} "
                          f"({anomaly['details']})")

            print(f"[扫描完成] 耗时: {elapsed:.0f}ms, "
                  f"异常: {total_anomalies} 只, "
                  f"异常池: {len(self.anomaly_watchlist)} 只")

            # 等待下一次扫描
            time.sleep(interval)

    def _detect_anomalies(self, quotes):
        """检测异常股票"""
        anomalies = []

        for symbol, quote in quotes.items():
            # 1. 涨停/跌停检测
            change_pct = (
                (quote['lastPrice'] - quote['lastClose']) / quote['lastClose'] * 100
            )

            if change_pct >= 9.9:
                anomalies.append({
                    'symbol': symbol,
                    'type': '涨停',
                    'details': f"涨{change_pct:.2f}%"
                })

            elif change_pct <= -9.9:
                anomalies.append({
                    'symbol': symbol,
                    'type': '跌停',
                    'details': f"跌{change_pct:.2f}%"
                })

            # 2. 放量检测
            # 成交额 > 10亿
            if quote['amount'] > 1000000000:
                anomalies.append({
                    'symbol': symbol,
                    'type': '放量',
                    'details': f"成交额{quote['amount']/1e8:.2f}亿"
                })

            # 3. 换手率检测（需要流通股本数据）
            # ...

        return anomalies

    def get_anomaly_watchlist(self):
        """获取异常股票池（供Level 2使用）"""
        return list(self.anomaly_watchlist)
```

## Level 2: 重点监控器

### 设计

```python
class FocusedMonitor:
    """重点股票监控器"""

    def __init__(self, max_subscriptions=500):
        self.max_subscriptions = max_subscriptions
        self.focused_watchlist = set()  # 重点股票池
        self.previous_quotes = {}  # 前一次快照

        # 初始化订阅
        self._init_subscriptions()

    def _init_subscriptions(self):
        """初始化订阅（加载自选股）"""
        # 从配置/数据库加载用户自选股
        my_watchlist = self._load_my_watchlist()

        for symbol in my_watchlist:
            self.add_to_focused(symbol)

    def add_to_focused(self, symbol):
        """添加到重点监控"""
        if len(self.focused_watchlist) < self.max_subscriptions:
            # 订阅
            xtdata.subscribe_quote(symbol, period='1d')
            self.focused_watchlist.add(symbol)

            print(f"[添加] {symbol} -> 重点监控 "
                  f"({len(self.focused_watchlist)}/{self.max_subscriptions})")
            return True
        else:
            print(f"[警告] 订阅已满({self.max_subscriptions})，无法添加 {symbol}")
            return False

    def remove_from_focused(self, symbol):
        """从重点监控移除"""
        if symbol in self.focused_watchlist:
            self.focused_watchlist.remove(symbol)
            if symbol in self.previous_quotes:
                del self.previous_quotes[symbol]

            print(f"[移除] {symbol} <- 重点监控 "
                  f"({len(self.focused_watchlist)}/{self.max_subscriptions})")

    def monitor_focused(self, interval=3):
        """
        高频监控重点股票

        Args:
            interval: 刷新间隔（秒），默认3秒
        """
        print(f"[启动] 重点监控器，股票数: {len(self.focused_watchlist)}")

        while True:
            if self.focused_watchlist:
                # 获取快照（超快！）
                start = time.time()
                quotes = xtdata.get_full_tick(list(self.focused_watchlist))
                elapsed = (time.time() - start) * 1000

                # 检测变化
                changes = self._detect_changes(quotes)

                if changes:
                    for change in changes:
                        print(f"[变化] {change}")

                print(f"[刷新] 耗时: {elapsed:.2f}ms, "
                      f"股票数: {len(self.focused_watchlist)}")

            time.sleep(interval)

    def _detect_changes(self, quotes):
        """检测实时变化（对比前后快照）"""
        changes = []

        for symbol, quote in quotes.items():
            if symbol in self.previous_quotes:
                previous = self.previous_quotes[symbol]

                # 价格变化
                price_change = quote['lastPrice'] - previous['lastPrice']
                price_change_pct = (price_change / previous['lastPrice']) * 100

                # 3秒内涨超过1%
                if abs(price_change_pct) > 1:
                    changes.append({
                        'symbol': symbol,
                        'type': '快速变化',
                        'details': f"{price_change_pct:+.2f}% (3秒内)"
                    })

                # 成交量变化
                volume_increase = quote['volume'] - previous['volume']
                if volume_increase > 0:
                    changes.append({
                        'symbol': symbol,
                        'type': '成交',
                        'details': f"+{volume_increase}手"
                    })

            # 保存当前快照
            self.previous_quotes[symbol] = quote

        return changes

    def _load_my_watchlist(self):
        """加载用户自选股（从配置/数据库）"""
        # TODO: 从数据库或配置文件加载
        # 示例：
        return ['600519.SH', '000001.SZ', '600036.SH']
```

## 完整系统：两级联动

```python
class TwoLevelMarketMonitor:
    """两级市场监控系统"""

    def __init__(self):
        # Level 1: 全A扫描器
        self.scanner = FullMarketScanner()

        # Level 2: 重点监控器
        self.monitor = FocusedMonitor(max_subscriptions=500)

        # 启动两个线程
        import threading

        self.scanner_thread = threading.Thread(
            target=self.scanner.scan_market,
            kwargs={'interval': 60},  # 60秒扫描一次
            daemon=True
        )

        self.monitor_thread = threading.Thread(
            target=self.monitor.monitor_focused,
            kwargs={'interval': 3},  # 3秒刷新一次
            daemon=True
        )

    def start(self):
        """启动监控系统"""
        print("=" * 80)
        print("两级市场监控系统启动")
        print("=" * 80)
        print()

        print("[Level 1] 全A扫描器")
        print(f"  - 监控范围: {len(self.scanner.all_stocks)} 只股票")
        print(f"  - 扫描频率: 60秒/次")
        print(f"  - 作用: 发现异常（涨停、跌停、放量）")
        print()

        print("[Level 2] 重点监控器")
        print(f"  - 监控数量: {len(self.monitor.focused_watchlist)}/500")
        print(f"  - 刷新频率: 3秒/次")
        print(f"  - 作用: 密切关注自选股 + 异常股")
        print()

        # 定期将异常股票添加到重点监控
        self._sync_thread = threading.Thread(
            target=self._sync_anomalies,
            daemon=True
        )

        # 启动所有线程
        self.scanner_thread.start()
        self.monitor_thread.start()
        self._sync_thread.start()

        print("[系统] 所有线程已启动")
        print()

    def _sync_anomalies(self):
        """定期同步异常股票到重点监控"""
        import time

        while True:
            # 每30秒同步一次
            time.sleep(30)

            # 获取异常股票池
            anomalies = self.scanner.get_anomaly_watchlist()

            # 添加到重点监控
            added_count = 0
            for symbol in anomalies:
                if symbol not in self.monitor.focused_watchlist:
                    success = self.monitor.add_to_focused(symbol)
                    if success:
                        added_count += 1

            if added_count > 0:
                print(f"[同步] 添加 {added_count} 只异常股票到重点监控")
```

## 优势总结

### ✅ 解决订阅限制

| 方案 | 监控范围 | 订阅需求 | 可行性 |
|------|---------|---------|--------|
| 单级监控（全订阅） | 5186只 | 5186个订阅 | ❌ 超限 |
| **两级监控** | 5186只 | 300-500个订阅 | ✅ 可行 |

### ✅ 性能优化

| Level | 频率 | 耗时 | CPU占用 |
|-------|------|------|---------|
| L1（扫描） | 60秒 | 550ms | < 1% |
| L2（监控） | 3秒 | 10ms | < 1% |
| **总计** | - | - | **< 2%** |

### ✅ 智能分级

- **全市场**: 不漏掉任何异常（涨停、跌停、放量）
- **重点池**: 密切关注自选股 + 异常股
- **动态调整**: 异常股票自动加入重点监控
