# 非交易时间数据策略

> 问题：非交易时间用户看不到股票列表怎么办？
> 创建日期: 2026-02-05

---

## 问题分析

### 交易时间 vs 非交易时间

**交易时间**:
- 周一到周五
- 上午：9:30-11:30
- 下午：13:00-15:00

**非交易时间**:
- 周末（周六、周日）
- 法定节假日
- 盘前（9:30前）
- 盘后（15:00后）
- 午休（11:30-13:00）

from datetime import datetime
import pandas as pd

class SmartQuoteService:
    """智能行情服务：处理交易时间和非交易时间"""

    def is_trading_time(self) -> bool:
        """判断是否交易时间"""
        now = datetime.now()
        current_time = now.strftime('%H%M%S')
        weekday = now.weekday()  # 0=周一, 6=周日

        # 周末不是交易时间
        if weekday >= 5:  # 周六、周日
            return False

        # 交易时间段
        morning_start = '093000'
        morning_end = '113000'
        afternoon_start = '130000'
        afternoon_end = '150000'

        is_morning = morning_start <= current_time <= morning_end
        is_afternoon = afternoon_start <= current_time <= afternoon_end

        return is_morning or is_afternoon

    def get_stock_list(self, symbols):
        """获取股票列表（智能处理非交易时间）"""

        if self.is_trading_time():
            # ✅ 交易时间：获取实时行情
            return self._get_realtime_quotes(symbols)
        else:
            # ⚠️ 非交易时间：使用历史数据最新一天
            return self._get_last_close(symbols)

    def _get_realtime_quotes(self, symbols):
        """获取实时行情（交易时间）"""
        import xtdata

        # 优先从订阅缓存
        try:
            data = xtdata.get_full_tick(symbols)
            if data and len(data) > 0:
                return self._format_quotes(data, source='realtime')
        except Exception as e:
            logger.warning(f"获取实时行情失败: {e}")

        # 降级：在线获取
        return self._get_online_quotes(symbols)

    def _get_last_close(self, symbols):
        """获取最后收盘价（非交易时间）"""
        import xtdata

        quotes = {}
        for symbol in symbols:
            try:
                # 获取最新日K线
                kline_data = xtdata.get_market_data_ex(
                    field_list=['open', 'high', 'low', 'close', 'volume', 'amount'],
                    stock_list=[symbol],
                    period='1d',
                    count=1  # 只获取最新一天
                )

                if symbol in kline_data:
                    df = kline_data[symbol]
                    if len(df) > 0:
                        last_row = df.iloc[-1]

                        # 转换为行情格式
                        quotes[symbol] = {
                            'lastPrice': float(last_row['close']),
                            'open': float(last_row['open']),
                            'high': float(last_row['high']),
                            'low': float(last_row['low']),
                            'volume': int(last_row['volume']),
                            'amount': float(last_row['amount']),
                            'lastClose': float(last_row['close']),  # 收盘价
                            'change': 0,  # 当天无变化
                            'changePct': 0,
                            'source': 'history',
                            'updateTime': df.index[-1]  # 历史日期
                        }

            except Exception as e:
                logger.error(f"获取{symbol}历史数据失败: {e}")

        return self._format_quotes(quotes, source='history')

    def _format_quotes(self, quotes, source):
        """格式化行情数据"""
        return {
            'data': quotes,
            'source': source,  # 'realtime' 或 'history'
            'update_time': datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        }
```

class QuoteStatus:
    """行情状态"""

    def __init__(self, quotes):
        self.quotes = quotes
        self.status = self._determine_status()

    def _determine_status(self):
        """判断行情状态"""
        # 检查是否有实时数据
        has_realtime = any(
            q.get('source') == 'realtime'
            for q in self.quotes.values()
        )

        if has_realtime:
            return {
                'code': 'trading',
                'message': '交易中',
                'color': 'green',
                'icon': '🟢'
            }
        else:
            return {
                'code': 'closed',
                'message': '已收盘',
                'color': 'gray',
                'icon': '⚪'
            }

    def to_dict(self):
        """返回带状态的数据"""
        return {
            'status': self.status,
            'quotes': self.quotes,
            'timestamp': datetime.now().isoformat()

class SmartQuoteService:
    """智能行情服务（推荐实现）"""

    def get_quotes(self, symbols):
        """获取股票行情（自动处理交易时间）"""

        # 1. 检测交易时间
        if self.is_trading_time():
            # 交易时间：优先实时
            return self._get_realtime(symbols)
        else:
            # 非交易时间：使用历史数据
            return self._get_history(symbols)

    def is_trading_time(self):
        """判断是否交易时间"""
        now = datetime.now()

        # 1. 检查是否周末
        if now.weekday() >= 5:
            return False

        # 2. 检查是否节假日
        if self.is_holiday(now):
            return False

        # 3. 检查是否交易时段
        current_time = now.strftime('%H%M%S')
        return '093000' <= current_time <= '150000'
```
