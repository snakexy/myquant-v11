## 聚合场景分类

### 1. K线聚合（低周期 → 高周期）

**场景**: 将分钟K线聚合成日K线、周K线等

```python
def aggregate_kline(data, from_period, to_period):
    """
    K线聚合：低周期 → 高周期

    示例：
    - 1分钟 → 5分钟
    - 5分钟 → 30分钟
    - 1分钟 → 1天
    """
    import pandas as pd

    # 确保索引是时间类型
    if not isinstance(data.index, pd.DatetimeIndex):
        data.index = pd.to_datetime(data.index)

    # 根据目标周期设置重采样规则
    period_map = {
        '1m': '1T',    # 1分钟
        '5m': '5T',
        '15m': '15T',
        '30m': '30T',
        '60m': '60T',  # 1小时
        '1d': '1D',    # 1天
        '1w': '1W',    # 1周
        '1M': '1M'     # 1月
    }

    rule = period_map.get(to_period, '1D')

    # OHLCV聚合
    aggregated = data.resample(rule).agg({
        'open': 'first',      # 开盘：第一个
        'high': 'max',        # 最高：最大值
        'low': 'min',         # 最低：最小值
        'close': 'last',      # 收盘：最后一个
        'volume': 'sum',      # 成交量：累加
        'amount': 'sum'       # 成交额：累加
    })

    # 删除NaN行（非交易时间）
    aggregated = aggregated.dropna()

    return aggregated
```

**性能**:
- 1000条1分钟 → 1天K线: **< 10ms**
- 本地聚合，无需网络请求

### 2. 分笔数据聚合（tick → 分钟K线）

**场景**: 将分笔数据聚合成分钟K线

```python
def aggregate_tick_to_kline(tick_data, period='1m'):
    """
    分笔数据聚合为分钟K线

    输入：tick数据（每笔交易）
    输出：分钟K线
    """
    import pandas as pd

    # 确保有时间索引
    if 'time' in tick_data.columns:
        tick_data = tick_data.set_index('time')

    if not isinstance(tick_data.index, pd.DatetimeIndex):
        tick_data.index = pd.to_datetime(tick_data.index)

    # 按分钟重采样
    period_map = {'1m': '1T', '5m': '5T'}
    rule = period_map.get(period, '1T')

    kline = tick_data.resample(rule).agg({
        'lastPrice': ['first', 'max', 'min', 'last'],  # OHLC
        'volume': 'sum',
        'amount': 'sum'
    })

    # 展平多级列名
    kline.columns = ['open', 'high', 'low', 'close', 'volume', 'amount']

    return kline.dropna()
```

**性能**:
- 10000笔 → 1分钟K线: **< 50ms**
- 适合生成分时图数据

### 3. 历史快照聚合（L2统计指标）

**场景**: 计算技术指标和统计摘要

```python
def calculate_indicators(kline_data):
    """
    计算技术指标

    输入：K线数据
    输出：包含技术指标的数据
    """
    import pandas as pd

    # 移动平均线
    kline_data['MA5'] = kline_data['close'].rolling(window=5).mean()
    kline_data['MA10'] = kline_data['close'].rolling(window=10).mean()
    kline_data['MA20'] = kline_data['close'].rolling(window=20).mean()
    kline_data['MA30'] = kline_data['close'].rolling(window=30).mean()

    # 涨跌幅
    kline_data['change_pct'] = (kline_data['close'] - kline_data['open']) / kline_data['open'] * 100

    # 振幅
    kline_data['amplitude'] = (kline_data['high'] - kline_data['low']) / kline_data['low'] * 100

    return kline_data


def calculate_summary(kline_data):
    """
    计算统计摘要（L2快照）

    输入：30天K线数据
    输出：统计摘要
    """
    summary = {
        'kline_30d': {
            'open': float(kline_data['open'].iloc[0]),
            'high': float(kline_data['high'].max()),
            'low': float(kline_data['low'].min()),
            'close': float(kline_data['close'].iloc[-1]),
            'volume': int(kline_data['volume'].sum()),
        },
        'indicators': {
            'MA5': float(kline_data['close'].tail(5).mean()),
            'MA10': float(kline_data['close'].tail(10).mean()),
            'MA20': float(kline_data['close'].tail(20).mean()),
            'MA30': float(kline_data['close'].tail(min(30, len(kline_data))).mean()),
        },
        'statistics': {
            'up_days': int(len(kline_data[kline_data['close'] > kline_data['open']])),
            'down_days': int(len(kline_data[kline_data['close'] < kline_data['open']])),
            'total_days': int(len(kline_data)),
            'up_ratio': float(len(kline_data[kline_data['close'] > kline_data['open']]) / len(kline_data) * 100)
        }
    }

    return summary
```

**性能**:
- 计算指标: **< 20ms**
- 计算摘要: **< 10ms**

### 4. 多股票聚合（批量统计）

**场景**: 获取多股票的涨跌排名、热点统计

```python
def aggregate_multi_stock_quotes(quotes_data):
    """
    多股票行情聚合

    输入：多股票实时行情
    输出：市场统计
    """
    import pandas as pd

    # 转换为DataFrame
    df = pd.DataFrame(quotes_data).T

    # 计算涨跌幅
    df['change_pct'] = (df['lastPrice'] - df['lastClose']) / df['lastClose'] * 100

    # 市场统计
    stats = {
        'total': len(df),
        'up_count': int(len(df[df['change_pct'] > 0])),
        'down_count': int(len(df[df['change_pct'] < 0])),
        'flat_count': int(len(df[df['change_pct'] == 0])),
        'avg_change_pct': float(df['change_pct'].mean()),
        'limit_up_count': int(len(df[df['change_pct'] >= 9.9])),  # 涨停
        'limit_down_count': int(len(df[df['change_pct'] <= -9.9])),  # 跌停
    }

    # 涨跌排名
    top_gainers = df.nlargest(10, 'change_pct')[['change_pct']].to_dict()['index']
    top_losers = df.nsmallest(10, 'change_pct')[['change_pct']].to_dict()['index']

    return {
        'statistics': stats,
        'top_gainers': top_gainers,
        'top_losers': top_losers
    }
```

**性能**:
- 100只股票聚合: **< 50ms**
- 1000只股票聚合: **< 200ms**

## 性能优化策略

### 1. 缓存聚合结果

```python
class AggregationCache:
    """聚合结果缓存"""

    def __init__(self):
        self.cache = {}

    def get_aggregated(self, symbol, period, count, agg_func):
        # 生成缓存key
        cache_key = f"{symbol}_{period}_{count}_{agg_func.__name__}"

        # 检查缓存
        if cache_key in self.cache:
            return self.cache[cache_key]

        # 执行聚合
        raw_data = get_kline(symbol, period, count)
        aggregated = agg_func(raw_data)

        # 缓存结果
        self.cache[cache_key] = aggregated
        return aggregated
```

### 2. 增量聚合

```python
def incremental_aggregation(new_data, old_aggregated=None):
    """
    增量聚合：只聚合新增数据

    适用场景：实时更新的数据
    """
    if old_aggregated is None:
        return aggregate_kline(new_data)

    # 只聚合新增部分
    new_timestamps = set(new_data.index) - set(old_aggregated.index)
    new_rows = new_data.loc[new_timestamps]

    # 合并
    combined = pd.concat([old_aggregated, new_rows])

    return combined
```

### 3. 并行聚合

```python
from concurrent.futures import ProcessPoolExecutor

def parallel_aggregate(symbols, period, count):
    """
    并行聚合多只股票的数据
    """
    with ProcessPoolExecutor(max_workers=4) as executor:
        results = executor.map(
            lambda s: calculate_summary(get_kline(s, period, count)),
            symbols
        )

    return dict(zip(symbols, results))
```

## 注意事项

### 1. 复权数据聚合

⚠️ **聚合前确认复权类型一致**
```python
# 错误：混合复权类型
data_none = get_kline(dividend_type='none')
data_front = get_kline(dividend_type='front')
combined = pd.concat([data_none, data_front])  # ❌ 错误！

# 正确：统一复权类型
data = get_kline(dividend_type='front')
aggregated = aggregate_kline(data, '1m', '1d')
```

### 2. 交易时间处理

⚠️ **删除非交易时间的聚合结果**
```python
# 删除非交易时间段
trading_hours = aggregated.between_time('09:30', '15:00')
```

### 3. 数据完整性验证

⚠️ **聚合后验证数据**
```python
# 验证OHLC逻辑
assert all(data['low'] <= data['high'])
assert all(data['open'] <= data['high'])
assert all(data['close'] >= data['low'])
```

## 总结

**聚合策略选择**:

| 场景 | 推荐方案 | 理由 |
|------|---------|------|
| 标准周期转换 | **直接获取目标周期** | 服务器端聚合更准确 |
| 非标准周期 | **客户端聚合** | 灵活但需要验证 |
| 分笔 → 分钟 | **客户端聚合** | XtQuant不提供 |
| 技术指标 | **客户端计算** | 支持自定义指标 |
| 批量统计 | **客户端聚合** | 需要跨股票计算 |

**性能建议**:
- ✅ 优先使用服务器提供的周期
- ✅ 缓存聚合结果
- ✅ 增量聚合实时数据
- ✅ 并行处理多股票
