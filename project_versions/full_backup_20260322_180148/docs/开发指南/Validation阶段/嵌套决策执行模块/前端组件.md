## 📊 状态管理

### Pinia Store

**路径**: `store/nestedBacktest.ts`

```typescript
import { defineStore } from 'pinia';

export const useNestedBacktestStore = defineStore('nestedBacktest', {
  state: () => ({
    // 任务列表
    tasks: [] as NestedBacktestTask[],

    // 当前任务
    currentTask: null as NestedBacktestTask | null,

    // 策略级别配置
    strategyLevels: [] as StrategyLevel[],

    // UI状态
    loading: false,
    activeTab: 'config',

    // 执行监控
    executionStatus: null as ExecutionStatus | null,
  }),

  getters: {
    // 获取运行中的任务
    runningTasks: (state) => {
      return state.tasks.filter(t => t.status === 'running');
    },

    // 获取已完成的任务
    completedTasks: (state) => {
      return state.tasks.filter(t => t.status === 'completed');
    },

    // 获取当前配置的级别数量
    levelCount: (state) => {
      return state.strategyLevels.length;
    },
  },

  actions: {
    // 创建嵌套回测任务
    async createBacktest(config: BacktestConfig) {
      this.loading = true;
      try {
        const response = await api.createNestedBacktest(config);
        this.tasks.push(response.data);
        return response.data;
      } finally {
        this.loading = false;
      }
    },

    // 获取回测结果
    async getResults(taskId: string) {
      this.loading = true;
      try {
        const response = await api.getNestedBacktestResults(taskId);
        this.currentTask = response.data;
        return response.data;
      } finally {
        this.loading = false;
      }
    },

    // 停止回测任务
    async stopTask(taskId: string) {
      const response = await api.stopNestedBacktest(taskId);
      const index = this.tasks.findIndex(t => t.task_id === taskId);
      if (index !== -1) {
        this.tasks[index].status = 'stopped';
      }
      return response.data;
    },

    // 添加策略级别
    addStrategyLevel(level: StrategyLevel) {
      this.strategyLevels.push(level);
    },

    // 移除策略级别
    removeStrategyLevel(levelIndex: number) {
      this.strategyLevels.splice(levelIndex, 1);
    },

    // 更新策略级别
    updateStrategyLevel(levelIndex: number, level: StrategyLevel) {
      this.strategyLevels[levelIndex] = level;
    },
  },
});
```

```vue
<script setup lang="ts">
import { ref } from 'vue';
import { useNestedBacktestStore } from '@/store/nestedBacktest';
import { createNestedBacktest } from '@/api/nested';

const nestedBacktestStore = useNestedBacktestStore();

const strategyLevels = ref([
  {
    level: 1,
    name: '日频选股',
    frequency: 'day',
    strategy: {
      class: 'TopkDropoutStrategy',
      parameters: { topk: 50, n_drop: 5 }
    }
  },
  {
    level: 2,
    name: '日内择时',
    frequency: '30min',
    strategy: {
      class: 'IntradayTimingStrategy',
      parameters: { entry_threshold: 0.02 }
    }
  }
]);

const backtestConfig = ref({
  start_time: '2024-01-01',
  end_time: '2024-12-31',
  account: 1000000,
  exchange: 'SHSZ'
});

const handleCreateBacktest = async () => {
  const response = await createNestedBacktest({
    strategy_levels: strategyLevels.value,
    backtest_config: backtestConfig.value,
    execution_config: {
      enable_interaction: true,
      joint_optimization: true
    }
  });

  console.log('任务创建成功:', response.data.task_id);
};
</script>
```

### 示例2：查看策略交互分析

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { analyzeStrategyInteraction } from '@/api/nested';

const interactionAnalysis = ref(null);

const analyzeInteraction = async () => {
  const response = await analyzeStrategyInteraction({
    strategy_levels: [
      {
        level: 1,
        name: '日频选股',
        strategy: { class: 'TopkDropoutStrategy' }
      },
      {
        level: 2,
        name: '日内择时',
        strategy: { class: 'IntradayTimingStrategy' }
      }
    ],
    analysis_config: {
      metrics: ['return', 'turnover', 'slippage'],
      sensitivity_analysis: true,
      ablation_study: true
    }
  });

  interactionAnalysis.value = response.data;
};

onMounted(() => {
  analyzeInteraction();
});
</script>
```
