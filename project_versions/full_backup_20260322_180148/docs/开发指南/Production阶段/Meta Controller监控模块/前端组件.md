## 🔄 API调用封装

```typescript
// frontend/src/api/meta-controller.ts
export const metaControllerApi = {
  // 获取性能监控
  getPerformance: (modelId?: string) => {
    return request.get('/api/v1/production/meta/monitor/performance', {
      params: { model_id: modelId }
    });
  },

  // 获取性能趋势
  getTrend: (modelId: string, period: string) => {
    return request.get(`/api/v1/production/meta/monitor/trend/${modelId}`, {
      params: { period }
    });
  },

  // 配置自动切换
  configureSwitch: (config: SwitchConfig) => {
    return request.post('/api/v1/production/meta/switch/configure', config);
  },

  // 触发切换
  triggerSwitch: (data: { from_model: string; to_model: string; reason: string }) => {
    return request.post('/api/v1/production/meta/switch/trigger', data);
  },

  // 创建A/B测试
  createABTest: (test: ABTestConfig) => {
    return request.post('/api/v1/production/meta/abtest/create', test);
  },

  // 获取市场状态
  getMarketState: () => {
    return request.get('/api/v1/production/meta/market/state');
  }
};
```

## 📊 WebSocket实时通信

```typescript
// Meta Controller WebSocket事件
interface MetaWebSocketEvents {
  'performance:update': PerformanceUpdate;
  'switch:triggered': SwitchEvent;
  'alert:created': AlertEvent;
  'abtest:completed': TestCompletedEvent;
}

class MetaWebSocket extends WebSocketClient<MetaWebSocketEvents> {
  subscribePerformance(modelId: string) {
    this.send({ type: 'subscribe', channel: 'performance', model_id: modelId });
  }

  subscribeAlerts() {
    this.send({ type: 'subscribe', channel: 'alerts' });
  }
}
```
