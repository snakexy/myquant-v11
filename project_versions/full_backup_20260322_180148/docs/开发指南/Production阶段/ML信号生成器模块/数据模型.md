## 📊 核心数据模型

### 1. ML预测输入 (MLPrediction)

```python
class MLPrediction(BaseModel):
    """ML预测输入"""
    instrument: str = Field(..., description="股票代码")
    prediction_score: float = Field(..., ge=0, le=1, description="ML预测评分 0-1")
    prediction: str = Field(..., description="预测方向: up/down/neutral")
    confidence: float = Field(..., ge=0, le=1, description="预测置信度")
    model_id: str = Field(..., description="模型ID")
    features_used: List[str] = Field(default=[], description="使用的特征列表")
```

### 2. 交易信号 (Signal)

```python
class Signal(BaseModel):
    """交易信号"""
    signal_id: str = Field(..., description="信号唯一标识")
    instrument: str = Field(..., description="股票代码")
    signal_type: SignalType = Field(..., description="信号类型: BUY/SELL/HOLD/WATCH")
    strength: SignalStrength = Field(..., description="信号强度: WEAK/NORMAL/STRONG")
    score: float = Field(..., description="ML评分")
    confidence: float = Field(..., description="置信度")
    current_price: float = Field(..., description="当前价格")
    target_price: float = Field(..., description="目标价格")
    reason: str = Field(..., description="信号生成原因")
    ml_model_id: str = Field(..., description="ML模型ID")
    timestamp: datetime = Field(..., description="生成时间")

class SignalType(str, Enum):
    BUY = "BUY"
    SELL = "SELL"
    HOLD = "HOLD"
    WATCH = "WATCH"

class SignalStrength(str, Enum):
    WEAK = "WEAK"
    NORMAL = "NORMAL"
    STRONG = "STRONG"
```

### 3. 持仓信息 (PositionInfo)

```python
class PositionInfo(BaseModel):
    """持仓信息"""
    current_position: int = Field(..., description="当前持仓")
    avg_cost: float = Field(..., description="平均成本")
    unrealized_pnl: float = Field(default=0, description="未实现盈亏")
```

### 4. 风险指标 (RiskMetrics)

```python
class RiskMetrics(BaseModel):
    """风险指标"""
    level: RiskLevel = Field(..., description="风险等级: NORMAL/MEDIUM/HIGH")
    concentration_ratio: float = Field(..., description="预测集中度")
    up_count: int = Field(..., description="上涨预测数量")
    down_count: int = Field(..., description="下跌预测数量")
    avg_confidence: float = Field(..., description="平均置信度")
    recommendation: str = Field(..., description="风险建议")

class RiskLevel(str, Enum):
    NORMAL = "NORMAL"
    MEDIUM = "MEDIUM"
    HIGH = "HIGH"
```

### 5. 仓位计算结果 (PositionResult)

```python
class PositionResult(BaseModel):
    """仓位计算结果"""
    adjusted_position: int = Field(..., description="调整后仓位")
    adjustment_reason: str = Field(..., description="调整原因")
    risk_level: RiskLevel = Field(..., description="风险等级")
```

### 6. 风险调整建议 (RiskAdjustment)

```python
class RiskAdjustment(BaseModel):
    """风险调整建议"""
    action: str = Field(..., description="调整动作")
    max_position_pct: float = Field(..., description="最大仓位比例")
    target_diversification: bool = Field(..., description="是否目标分散化")
    reason: str = Field(..., description="调整原因")
```
